1. #include <stdio.h>
  2. #include <windows.h>
  3. #define LEN 25
  4. void HideCursor();
  5. void Gotoxy(int x, int y);
  6. void DrawBox();
  7. int main(void)
  8. {
  9. int len;
  10. HideCursor();
  11. for(len = 1; len <= LEN; len++){
  12. Gotoxy(2 * len, 1 );
  13. printf("█");
  14. Gotoxy(21, 4);
  15. printf("已完成%d%%", 4 * len);
  16. Sleep(100);
  17. }
  18. Gotoxy(21, 4);
  19. printf("下载已完成");
  20. return 0;
  21. }
  22. //隐藏光标
  23. void HideCursor()
  24. {
  25. CONSOLE_CURSOR_INFO cursor_info = {1, 0};//后边的0代表光标不可见
  26. SetConsoleCursorInfo(GetStdHandle(STD_OUTPUT_HANDLE), &cursor_info);
  27. }
  28. //光标移动
  29. void Gotoxy(int x, int y)
  30. {
  31. HANDLE hout; //定义句柄变量hout
  32. COORD coord; //定义结构体coord
  33. coord.X = x;
  34. coord.Y = y;
  35. hout = GetStdHandle(STD_OUTPUT_HANDLE);//获得标准输出（屏幕）句柄
  36. SetConsoleCursorPosition(hout, coord);//移动光标
  37. }
  38. //画边框
  39. void DrawBox()
  40. {
  41. printf("╔═════════════════════════╗");
  42. putchar('\n');
  43. printf("║ ║");
  44. putchar('\n');
  45. printf("╚═════════════════════════╝");
  46. }
